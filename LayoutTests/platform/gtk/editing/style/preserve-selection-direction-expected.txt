Before foreColor 1:
| "Thi<#selection-anchor>s is <#selection-focus>some sample text"

After foreColor 1:
| "Thi<#selection-anchor>"
| <font>
|   color="#008000"
|   "s is <#selection-focus>"
| "some sample text"

After foreColor 2:
| "Thi<#selection-focus>"
| <font>
|   color="#008000"
