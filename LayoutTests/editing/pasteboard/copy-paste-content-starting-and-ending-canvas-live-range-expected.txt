| "\n"
| <#selection-anchor>
| <canvas>
|   height="100"
|   style="border: 1px solid black"
|   width="100"
| "\nsome text\n"
| <canvas>
|   height="100"
|   style="border: 1px solid black"
|   width="100"
| "\nsome more text\n"
| <canvas>
|   height="100"
|   style="border: 1px solid black"
|   width="100"
| <#selection-focus>
| "\n"
