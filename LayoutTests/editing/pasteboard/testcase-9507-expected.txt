After copy:
| "
"
| <div>
|   class="editing"
|   id="test"
|   "
<#selection-anchor>foo
"
|   <div>
|     style="color: rgb(255, 0, 0);"
|     "
"
|     <div>
|       "bar<#selection-focus>"
|     "
"
|   "
"
| "
"

After paste:

"

|   class="editing"
|   id="test"
|   "foo"
|   <div>
|     style="color: rgb(255, 0, 0); "
|     "bar<#selection-caret>"
|   <div>
|     style="color: rgb(255, 0, 0);"
|     "
"
|   "
