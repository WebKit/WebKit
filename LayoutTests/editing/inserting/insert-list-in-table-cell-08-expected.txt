| <table>
|   border="1"
|   id="element"
|   <tbody>
|     <tr>

|         "fsdf"
|       <td>
|         "fsdf"
|     <tr>
|       <td>
|         "gghfg"
|       <td>
|         "fsfg"
| <#selection-focus>

After:

| <table>
|   border="1"
|   id="element"
|   <tbody>

|       <td>
